    if(!file.exists(fname_ext)){
        FILE_URL <- paste0(BASE_URL, fname_ext)
        download.file(FILE_URL,

read_shp_from_zip <- function(shpfile) {
  td <- tempdir()
  zip_contents <- unzip(shpfile, exdir = td)
  shp <- zip_contents[grepl("\\.shp$", zip_contents)]
  us_sf <- read_sf(shp)
  return(us_sf) 
}
